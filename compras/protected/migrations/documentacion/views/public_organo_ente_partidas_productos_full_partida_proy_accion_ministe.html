SELECT
    organo_ente_partidas_productos_full_partida_proy_accion.ente_rector,
    organo_ente_partidas_productos_full_partida_proy_accion.ente_organo,
    organo_ente_partidas_productos_full_partida_proy_accion.descipcion_partida,
    organo_ente_partidas_productos_full_partida_proy_accion.cod_partida,
    organo_ente_partidas_productos_full_partida_proy_accion.monto_presupuesto,
    organo_ente_partidas_productos_full_partida_proy_accion.codigo_producto,
    organo_ente_partidas_productos_full_partida_proy_accion.producto,
    organo_ente_partidas_productos_full_partida_proy_accion.unidad,
    organo_ente_partidas_productos_full_partida_proy_accion.cantidad,
    organo_ente_partidas_productos_full_partida_proy_accion.costo_unitario,
    organo_ente_partidas_productos_full_partida_proy_accion.monto_total,
    organo_ente_partidas_productos_full_partida_proy_accion.ente_organo_id,
    organo_ente_partidas_productos_full_partida_proy_accion.proyectonom,
    organo_ente_partidas_productos_full_partida_proy_accion.accion_nom,
    organo_ente_partidas_productos_full_partida_proy_accion.codigo_accion
FROM organo_ente_partidas_productos_full_partida_proy_accion
WHERE (organo_ente_partidas_productos_full_partida_proy_accion.ente_rector::text IN ( SELECT entes_organos.nombre
FROM entes_organos
WHERE entes_organos.nombre::text ~~ 'Ministerio%'::text
        OR entes_organos.nombre::text ~~ 'MINISTERIO%'::text
        OR entes_organos.nombre::text ~~ 'Ministerio%'::text
        OR entes_organos.nombre::text ~~ 'Vicepresidencia%'::text))
ORDER BY organo_ente_partidas_productos_full_partida_proy_accion.ente_organo_id
