public.partidas_productos2(View)
Namepartidas_productos2
SQL 
SELECT
    a.partida_id,
    a.p1,
    a.p2,
    a.p3,
    a.p4,
    a.nombre,
    b.producto_id,
    b.tipo_operacion,
    b.partida_producto_id,
    c.nombre AS nombre_prod,
    c.cod_segmento,
    c.cod_familia,
    c.cod_clase,
    c.cod_producto
FROM partidas a
    JOIN partida_productos b ON a.partida_id = b.partida_id
    JOIN productos c ON b.producto_id = c.producto_id
