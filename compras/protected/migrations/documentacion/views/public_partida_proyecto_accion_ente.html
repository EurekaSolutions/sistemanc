public.partida_proyecto_accion_ente(View)

SQL 
SELECT
    a.presupuesto_partida_id,
    a.partida_id,
    a.ente_organo_id,
    b.proyecto_id,
    c.accion_id,
    c.codigo_accion
FROM trimestre2.presupuesto_partidas a LEFT
    JOIN trimestre2.presupuesto_partida_proyecto b ON a.presupuesto_partida_id = b.presupuesto_partida_id LEFT
    JOIN trimestre2.presupuesto_partida_acciones c ON a.presupuesto_partida_id = c.presupuesto_partida_id
