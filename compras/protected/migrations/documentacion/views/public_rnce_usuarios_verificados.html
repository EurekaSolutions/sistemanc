public.rnce_usuarios_verificados(View)
Namernce_usuarios_verificados
SQL 
SELECT
    u.usuario,
    u.nombre,
    u.cedula,
    e.nombre AS ente_organo,
    CASE WHEN u.correo_verificado = true THEN 'SI'::text WHEN u.correo_verificado = false THEN 'NO'::text ELSE 'NULO'::text END AS correo_verificado
FROM usuarios u
    JOIN entes_adscritos a USING (ente_organo_id)
    JOIN entes_organos e USING (ente_organo_id)
ORDER BY CASE WHEN u.correo_verificado = true THEN 'SI'::text WHEN u.correo_verificado = false THEN 'NO'::text ELSE 'NULO'::text END
