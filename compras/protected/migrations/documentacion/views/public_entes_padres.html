public.entes_padres(View)
Nameentes_padres
SQL 
SELECT
    a.padre_id,
    b.nombre AS ente_padre,
    a.ente_organo_id AS ente_organo_hijo_id,
    a.ente_adscrito_id
FROM entes_adscritos a FULL
    JOIN entes_organos b ON a.padre_id = b.ente_organo_id
ORDER BY a.padre_id
