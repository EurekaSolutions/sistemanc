trimestre1.presupuesto_aprobado_entes_organos(View)
Namepresupuesto_aprobado_entes_organos
SQL 
SELECT
    sum(b.monto_presupuestado) AS presupuesto_aprobado_rnce,
    b.ente_organo_id,
    c.nombre AS nombre_ente_organo
FROM rnce_partidas a
    JOIN trimestre1.presupuesto_partidas b USING (partida_id)
    JOIN entes_organos c USING (ente_organo_id)
GROUP BY b.ente_organo_id,
    c.nombre
ORDER BY sum(b.monto_presupuestado) DESC
