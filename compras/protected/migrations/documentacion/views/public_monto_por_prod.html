public.monto_por_prod(View)
Namemonto_por_prod
SQL 
SELECT
    sum(a.monto_presupuesto) AS monto_pretot,
    b.nombre,
    a.producto_id,
    b.cod_segmento,
    b.cod_familia,
    b.cod_clase,
    b.cod_producto
FROM presupuesto_productos a
    JOIN productos b ON a.producto_id = b.producto_id
GROUP BY a.producto_id,
    b.nombre,
    b.cod_segmento,
    b.cod_familia,
    b.cod_clase,
    b.cod_producto
ORDER BY sum(a.monto_presupuesto) DESC
