trimestre2.factura_proveedores(View)
Namefactura_proveedores
SQL 
SELECT
    a.rif,
    a.razon_social,
    a.estatus_contratista_descrip,
    a.proveedor_id,
    a.ente_organo_id,
    h.nombre AS ente_organo,
    b.num_factura,
    b.id AS id_factura,
    sum(c.costo_unitario * c.cantidad_adquirida::numeric + c.costo_unitario * c.cantidad_adquirida::numeric * g.porcentaje / 100::numeric) AS totfacura
FROM proveedores_entes_edoscontra a
    JOIN trimestre2.facturas b ON a.ente_organo_id = b.ente_organo_id
        AND a.proveedor_id = b.proveedor_id
    JOIN trimestre2.facturas_productos c ON c.factura_id = b.id
    JOIN iva g ON g.id = c.iva_id
    JOIN entes_organos h ON h.ente_organo_id = a.ente_organo_id
GROUP BY a.rif,
    a.razon_social,
    a.estatus_contratista_id,
    a.estatus_contratista_descrip,
    a.proveedor_id,
    h.nombre,
    a.ente_organo_id,
    b.num_factura,
    b.id
ORDER BY a.proveedor_id,
    a.ente_organo_id,
    b.num_factura
