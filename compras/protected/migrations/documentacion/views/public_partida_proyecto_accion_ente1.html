SQL 
SELECT
    a.presupuesto_partida_id,
    a.partida_id,
    p.nombre AS nom_partida,
    ((p.p1::text || p.p2::text) || p.p3::text) || p.p4::text AS cod_partida,
    a.ente_organo_id,
    b.proyecto_id,
    c.accion_id,
    c.codigo_accion,
    d.nombre AS proyectonom,
    f.nombre AS accion_nom
FROM presupuesto_partidas a
    JOIN partidas p ON p.partida_id = a.partida_id LEFT
    JOIN presupuesto_partida_proyecto b ON a.presupuesto_partida_id = b.presupuesto_partida_id LEFT
    JOIN presupuesto_partida_acciones c ON a.presupuesto_partida_id = c.presupuesto_partida_id LEFT
    JOIN proyectos d ON d.proyecto_id = b.proyecto_id LEFT
    JOIN acciones f ON f.accion_id = c.accion_id
