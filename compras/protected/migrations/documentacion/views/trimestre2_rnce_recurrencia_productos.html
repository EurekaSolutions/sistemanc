trimestre2.rnce_recurrencia_productos(View)
Namernce_recurrencia_productos
SQL 
SELECT
    count(a.producto_id) AS recurrencia,
    b.nombre AS producto,
    (((((b.cod_segmento::text || '-'::text) || b.cod_familia::text) || '-'::text) || b.cod_clase::text) || '-'::text) || b.cod_producto::text AS codigo_producto
FROM trimestre2.presupuesto_productos a
    JOIN productos b ON a.producto_id = b.producto_id
    JOIN trimestre2.presupuesto_partidas c ON a.proyecto_partida_id = c.presupuesto_partida_id
WHERE c.ente_organo_id <> ALL (ARRAY[12615::bigint,
    12616::bigint,
    12617::bigint,
    12618::bigint,
    12619::bigint])
GROUP BY a.producto_id,
    b.nombre,
    b.cod_segmento,
    b.cod_familia,
    b.cod_clase,
    b.cod_producto
ORDER BY count(a.producto_id) DESC
