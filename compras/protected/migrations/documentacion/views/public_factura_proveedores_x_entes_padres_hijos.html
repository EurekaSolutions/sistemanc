public.factura_proveedores_x_entes_padres_hijos(View)
Namefactura_proveedores_x_entes_padres_hijos
SQL 
SELECT
    a.padre_id,
    a.ente_padre,
    a.ente_organo_hijo_id,
    a.organo_hijo,
    b.rif,
    b.razon_social,
    b.estatus_contratista_descrip,
    b.proveedor_id,
    b.ente_organo_id,
    o.nombre AS ente_faturado,
    b.num_factura,
    b.id_factura,
    b.totfacura
FROM trimestre2.factura_proveedores b FULL
    JOIN entes_organos_ph a ON a.ente_organo_hijo_id = b.ente_organo_id FULL
    JOIN entes_organos o ON o.ente_organo_id = b.ente_organo_id
WHERE b.id_factura IS NOT NULL
ORDER BY a.padre_id,
    a.ente_padre,
    a.ente_organo_hijo_id,
    a.organo_hijo
