public.proveedores_entes_edoscontra(View)
Nameproveedores_entes_edoscontra
SQL 
SELECT
    a.rif,
    a.razon_social,
    a.estatus_contratista_id,
    CASE WHEN a.estatus_contratista_id = 1 THEN 'NO REGISTRADA EN EL RNC( NO Inscrita en el Registro Nacional de Contratistas,
    Proceso de registro sin finalizar)'::text WHEN a.estatus_contratista_id = 2 THEN 'CALIFICADA( HABILITADA para contratar con el Estado de conformidad con el Artículo 48 de LCP)'::text WHEN a.estatus_contratista_id = 3 THEN 'INHABILITADA PROVISIONALMENTE POR MEDIDA ADMINISTRATIVA (El Registro Nacional de Contratistas INHABILITA provisionalmente para la contratación con el Estado,
    por haberse iniciado un procedimiento administrativo.)'::text WHEN a.estatus_contratista_id = 4 THEN 'REGISTRADA NO ACTUALIZADA( INHABILITADA para contratar con el Estado de conformidad con el Articulo 50 LCP)'::text WHEN a.estatus_contratista_id = 5 THEN 'EN PROCESO DE DESCAPITALIZACION( INHABILITADA para contratar con el Estado por encontrarse en proceso de DESCAPITALIZACION según el Artículo 264 del Código de Comercio Venezolano.)'::text WHEN a.estatus_contratista_id = 6 THEN 'PERSONA JURIDICA CON MANEJO DE FONDOS PUBLICOS ( excluida de las modalidades de selección de conformidad con el Articulo 5,
    numerales 8,
    9 y 10 de la LCP)'::text WHEN a.estatus_contratista_id = 8 THEN 'SANCIONADA ( INHABILITADA para contratar con el Estado de conformidad con lo establecido en los Artículos 167 y 168 de LCP)'::text WHEN a.estatus_contratista_id = 10 THEN 'EXTRANJERA,
    SIN DOMICILIO NI FILIAL EN VENEZUELA ( exceptuada de la CALIFICACIÓN por el Registro Nacional de Contratistas de conformidad con el Articulo 49,
    numerales 6 y 7.)'::text WHEN a.estatus_contratista_id = 12 THEN 'INSCRITA Y ACTUALIZADA( HABILITADA para contratar con el Estado hasta 4.000 UT en Bienes y Servicios / 5.000 UT en Obras de conformidad con los Articulos 47 y 49 numeral 1,
    o por estar enmarcada en alguno de los numerales de Exclusión de Modalidad previstos en el Articulos 5 de la LCP)'::text WHEN a.estatus_contratista_id = 13 THEN 'Inscrita,
    Actualizada,
    No Calificada (No alcanzó la Calificación prevista en el Artículo 48 LCP HABILITADA para contratar con el Estado hasta 4.000 UT en Bienes y Servicios / 5.000 UT en Obras de conformidad con los Articulos 47 y 49 numeral 1 LCP)'::text WHEN a.estatus_contratista_id IS NULL THEN NULL::text ELSE NULL::text END AS estatus_contratista_descrip,
    b.proveedor_id,
    b.ente_organo_id
FROM proveedores a
    JOIN proveedores_entes_organos b ON a.id = b.proveedor_id
ORDER BY a.estatus_contratista_id
