trimestre1.presupuesto_importados_entes_organos(View)
Namepresupuesto_importados_entes_organos
SQL 
SELECT
    sum(a.monto_presupuesto * d.tasa * a.cantidad::numeric) AS presupuesto_importacion_rnce,
    b.ente_organo_id,
    c.nombre AS nombre_ente_organo
FROM trimestre1.presupuesto_importacion a
    JOIN trimestre1.presupuesto_partidas b ON a.presupuesto_partida_id = b.presupuesto_partida_id
    JOIN entes_organos c ON c.ente_organo_id = b.ente_organo_id
    JOIN tasas d ON d.divisa_id = a.divisa_id
GROUP BY b.ente_organo_id,
    c.nombre
ORDER BY sum(a.monto_presupuesto) DESC
