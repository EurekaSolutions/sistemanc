trimestre1.presupuesto_productos_entes_organos(View)
Namepresupuesto_productos_entes_organos
SQL 
SELECT
    sum(a.monto_presupuesto) AS presupuesto_productos_rnce,
    b.ente_organo_id,
    c.nombre AS nombre_ente_organo
FROM trimestre1.presupuesto_productos a
    JOIN trimestre1.presupuesto_partidas b ON a.proyecto_partida_id = b.presupuesto_partida_id
    JOIN entes_organos c USING (ente_organo_id)
GROUP BY b.ente_organo_id,
    c.nombre
ORDER BY sum(a.monto_presupuesto) DESC
