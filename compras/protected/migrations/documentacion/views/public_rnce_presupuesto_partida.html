SQL 
SELECT
    replace(round(sum(a.monto_presupuesto),
    2)::text,
    '.'::text,
    ',
    '::text) AS total_partida,
    (((((c.p1 || '-'::text) || lpad(c.p2::text,
    2,
    '0'::text)) || '-'::text) || lpad(c.p3::text,
    2,
    '0'::text)) || '-'::text) || lpad(c.p4::text,
    2,
    '0'::text) AS cod_presupuesto,
    c.nombre AS concepto_partida
FROM presupuesto_productos a
    JOIN presupuesto_partidas b ON a.proyecto_partida_id = b.presupuesto_partida_id
    JOIN rnce_partidas c USING (partida_id)
GROUP BY b.partida_id,
    c.p1,
    c.p2,
    c.p3,
    c.p4,
    c.nombre
ORDER BY b.partida_id
