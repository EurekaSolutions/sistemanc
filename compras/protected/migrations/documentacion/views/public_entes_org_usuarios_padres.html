public.entes_org_usuarios_padres(View)
Nameentes_org_usuarios_padres
SQL 
SELECT
    DISTINCT p.ente_padre,
    p.padre_id,
    a.ente_organo_id,
    a.nombre AS ente,
    a.compra,
    a.rendicion,
    b.correo,
    b.nombre AS usuario,
    b.ultima_visita_el,
    b.usuario_id,
    b.correo_verificado,
    b.creado_el
FROM entes_organos a
    JOIN usuarios b USING (ente_organo_id)
    JOIN entes_padres p ON a.ente_organo_id = p.padre_id
WHERE b.rol::text = ANY (ARRAY['organo'::character varying::text,
    'ente'::character varying::text])
ORDER BY p.padre_id
